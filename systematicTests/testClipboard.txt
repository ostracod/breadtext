
TEST "System copy character"
ADD_LINE "abcdef"
ADD_LINE "ghijkl"
ADD_LINE "mnopqr"
ADD_LINE ""
SET_POS 2 1
PRESS_KEY Y c
SET_POS 4 0
PRESS_KEY P
ASSERT_LINE_COUNT 4
ASSERT_LINE 1 "abcdef"
ASSERT_LINE 2 "ghijkl"
ASSERT_LINE 3 "mnopqr"
ASSERT_LINE 4 "h"

TEST "System copy selection"

ASSERT_LINE 2 "ghijkl"
ASSERT_LINE 3 "mnopqr"
ASSERT_LINE 4 "hij"

SET_POS 1 0
PRESS_KEY y H DOWN DOWN c
SET_POS 4 0
PRESS_KEY P
SET_POS 8 0
PRESS_KEY p

ASSERT_LINE 1 "        one"

ASSERT_LINE 3 "                three"
ASSERT_LINE 4 "    one"
ASSERT_LINE 5 "        two"
ASSERT_LINE 6 "            three"
ASSERT_LINE 7 "    four"
ASSERT_LINE 8 "        five"
ASSERT_LINE 9 "        one"
ASSERT_LINE 10 "            two"
ASSERT_LINE 11 "                three"
