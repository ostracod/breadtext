TEST "Replace"
ADD_LINE "abc def ghi"
ADD_LINE "def abc ghi"
ADD_LINE "abc ghi def"
TEXT_COMMAND "replace def bread"
ASSERT_LINE_COUNT 3
ASSERT_LINE 1 "abc bread ghi"
ASSERT_LINE 2 "bread abc ghi"
ASSERT_LINE 3 "abc ghi bread"
