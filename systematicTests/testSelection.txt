
TEST "Character select"
ADD_LINE "abcdefghi"
SET_POS 1 3
PRESS_KEY h RIGHT RIGHT RIGHT
ASSERT_POS 1 6
ASSERT_HIGHLIGHT_POS 1 3

TEST "Escape character select"
ADD_LINE "abcdefghi"
SET_POS 1 3
PRESS_KEY h RIGHT RIGHT RIGHT
PRESS_KEY ESC d
ASSERT_LINE_COUNT 1
ASSERT_LINE 1 "abcdefhi"

TEST "Line select"
ADD_LINE "abc"
ADD_LINE "defg"
ADD_LINE "hijkl"
SET_POS 2 0
PRESS_KEY H
ASSERT_POS 2 0
ASSERT_HIGHLIGHT_POS 2 4

TEST "Line select and move down"
ADD_LINE "abc"
ADD_LINE "defg"
ADD_LINE "hijkl"
SET_POS 2 0
PRESS_KEY H DOWN
ASSERT_POS 3 5
ASSERT_HIGHLIGHT_POS 2 0

TEST "Line select and move up"
ADD_LINE "abc"
ADD_LINE "defg"
ADD_LINE "hijkl"
SET_POS 2 0
PRESS_KEY H UP
ASSERT_POS 1 0
ASSERT_HIGHLIGHT_POS 2 4

TEST "Line select escape"
ADD_LINE "abc"
ADD_LINE "defg"
ADD_LINE "hijkl"
SET_POS 2 0
PRESS_KEY H ESC d
ASSERT_LINE_COUNT 3
ASSERT_LINE 1 "abc"
ASSERT_LINE 2 "efg"
ASSERT_LINE 3 "hijkl"

TEST "Escape word select"
ADD_LINE "abc def ghi"

PRESS_KEY w ESC d
