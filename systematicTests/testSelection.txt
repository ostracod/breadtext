
TEST "Character select"
ADD_LINE "abcdefghi"
SET_POS 1 3
PRESS_KEY h RIGHT RIGHT RIGHT
ASSERT_POS 1 6
ASSERT_HIGHLIGHT_POS 1 3

TEST "Escape character select"
ADD_LINE "abcdefghi"
SET_POS 1 3
PRESS_KEY h RIGHT RIGHT RIGHT
PRESS_KEY ESC d
ASSERT_LINE_COUNT 1
ASSERT_LINE 1 "abcdefhi"

TEST "Escape character select alternative"
ADD_LINE "abcdefghi"
SET_POS 1 3
PRESS_KEY h RIGHT RIGHT RIGHT
PRESS_KEY , d
ASSERT_LINE_COUNT 1
ASSERT_LINE 1 "abcdefhi"
